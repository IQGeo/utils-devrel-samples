Adding a Name Manager

A name managers is a server-side class that provides default names for objects as they are inserted. The class is activated by register it as a server-side trigger. The code below provides a simple example.

 

class StructNameManager():
    ##
    ## Engine to auto-name structures
    ##
    
    prefixes = { 'manhole':    'M',
                 'cabinet':    'C',
                 'pole':       'P',
                 'drop_point': 'DP' }

    @classmethod
    def registerTriggers(self,NetworkView):
        ##
        ## Register self's trigger methods on NETWORKVIEW
        ##

        NetworkView.registerTrigger('struct', 'pos_insert',self,'setNameFor')
        NetworkView.registerTrigger('struct', 'pos_update',self,'setNameFor')

        
    def __init__(self,nw_view,progress):
        ##
        ## Init slots of self
        ##
        ## NW_VIEW is a NetworkView. PROGRESS is a MywProgressHandler

        self.nw_view  = nw_view
        self.db_view  = nw_view.db_view
        self.progress = progress
         
    
    def setNameFor(self,struct,orig_struct=None):
        ##
        ## Sets name value for feature
        ##

        ft = struct.feature_type
        
        # Check for not auto-named
        prefix = self.prefixes.get(ft)
        if not prefix: return

        # Check for name already set
        if struct.name: return

        # Set name
        struct.name = '{}-{}'.format(prefix,struct._id)
        self.db_view.table(ft).update(struct)

# Register triggers
from myworldapp.modules.comms.server.api.network_view import NetworkView
StructNameManager.registerTriggers(NetworkView)

To enable the name manager during server startup, add the following line in the __init__.py of your module's server directory:

# Import name manager and register triggers
from .struct_name_manager import StructNameManager

To enable the name manager in custom scripts, simply import the file.
